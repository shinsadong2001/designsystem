// GNB 컴포넌트 사용 예시 (React + TypeScript)
import { GNB } from './components';

// 홈 기본
<GNB type="Home" subType="None" cartCount={3} onCart={handleCart} />

// 스크롤 내렸을 때
<GNB type="Home" subType="DownScroll" cartCount={3} />

// 서브 페이지 타이틀바
<GNB type="Sub" title="상품 상세" cartCount={1} onBack={handleBack} />

// 검색 모드
<GNB type="Search" subType="검색어 입력중"
     searchValue={query} onSearchChange={setQuery} />

// 비활성화
<GNB type="PDP" title="마이페이지" disabled />

  type="Home"
